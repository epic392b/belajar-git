command command github step by step:

1. mkdir proyek #bikin folder untuk repository nya
2. cd proyek #masuk ke folder repo nya
3. git init #buat repository .git di folder lokalnya
4. git branch --global init.defaultBranch main #set supaya brach defaultnya itu main bukan master
5. git status #liat status dari git
6. git commit -m "komentar"  #untuk commit fie git di komentar
7. git remote add origin https: #tambahkan remote https
8. git remote set-url origin ssh #untuk ubah https jadi ssh (disarankan)
9. git add origin ssh # sama kayak nomor 7 tapi versi ssh
10. git push -u origin main #untuk melakukan push pertama kali untuk menghubungkan dengan branch main
11. git add [file] #untuk massukan file ke staging area atau tempat persiapan untuk perubahan 
12. git push #buat simpan perubahan di github online
13. git pull #untuk mengambil perubahan dari github
